void build(const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)//建树
{
	if (Left == Right) 
	{
		Tree[TreeNode] = a[Left];
		return;
	}
	ll Mid = (Left+Right)>>1;
	build(Left, Mid, TreeNode<<1), build(Mid+1, Right,TreeNode<<1|1);
	Tree[TreeNode] = Tree[TreeNode<<1] + Tree[TreeNode<<1 | 1];
}
void Mxbuild(ll Left=1, ll Right=n, ll TreeNode=1)//建树
{
	if (Left == Right) 
	{
		Tree[TreeNode] = a[Left];
		return;
	}
	ll Mid = (Left+Right)>>1;
	Mxbuild(Left, Mid, TreeNode<<1), Mxbuild(Mid+1, Right,TreeNode<<1| 1);
	Tree[TreeNode] = max(Tree[TreeNode<<1] , Tree[TreeNode<<1|1]);
}
void update(const ll& RangeL, const ll& RangeR, const ll& Val, const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)//区间增加：l,r为增加区间，c为增加值
{
	if (RangeL <= Left && Right <= RangeR) 
	{
		Tree[TreeNode] += (Right - Left + 1) * Val;
		lazy[TreeNode] += Val;
		return;
	}
	ll Mid = (Left + Right) >>1;
	if (lazy[TreeNode] && Left != Right) 
	{
		Tree[TreeNode<<1] += lazy[TreeNode] * (Mid - Left + 1);
		Tree[TreeNode<<1| 1] += lazy[TreeNode] * (Right - Mid);
		lazy[TreeNode<<1] += lazy[TreeNode];
		lazy[TreeNode<<1| 1] += lazy[TreeNode];
		lazy[TreeNode] = 0;
	}
	if (RangeL <= Mid) update(RangeL, RangeR, Val, Left, Mid, TreeNode<<1);
	if (RangeR > Mid) update(RangeL, RangeR, Val, Mid + 1, Right, TreeNode<<1|1);
	Tree[TreeNode] = Tree[TreeNode<<1] + Tree[TreeNode<<1| 1];
}
ll getsum(const ll& RangeL,const ll& RangeR, const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)//区间增加的求和：l,r为增加区间
{
	if (RangeL <= Left && Right <= RangeR) return Tree[TreeNode];
	ll Mid = (Left + Right) >>1;
	if (lazy[TreeNode]) 
	{
		Tree[TreeNode<<1] += lazy[TreeNode] * (Mid - Left + 1);
		Tree[TreeNode<<1| 1] += lazy[TreeNode] * (Right - Mid);
		lazy[TreeNode<<1] += lazy[TreeNode], lazy[TreeNode<<1|1] += lazy[TreeNode];
		lazy[TreeNode] = 0;
	}
	ll sum = 0;
	if (RangeL <= Mid) sum = getsum(RangeL, RangeR, Left, Mid, TreeNode<<1);
	if (RangeR > Mid) sum += getsum(RangeL, RangeR, Mid + 1, Right, TreeNode<<1| 1);
	return sum;
}
ll bgetsum(const ll& RangeL, const ll& RangeR, const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)//区间值修改的求和：l,r为求和区间
{
	if (RangeL <= Left && Right <= RangeR) return Tree[TreeNode];
	ll m = (Left + Right) / 2;
	if (lazy[TreeNode]) 
	{
		Tree[TreeNode<<1] = lazy[TreeNode] * (m - Left + 1);
		Tree[TreeNode<<1| 1] = lazy[TreeNode] * (Right - m);
		lazy[TreeNode<<1] = lazy[TreeNode<<1| 1] = lazy[TreeNode];
		lazy[TreeNode] = 0;
	}
	ll sum = 0;
	if (RangeL <= m) sum = bgetsum(RangeL, RangeR, Left, m, TreeNode<<1);
	if (RangeR > m) sum += bgetsum(RangeL, RangeR, m + 1, Right, TreeNode<<1| 1);
	return sum;
}
void bupdate(const ll& RangeL, const ll& RangeR, const ll& Val, const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)//区间值修改：l,r为修改区间，c为修改目标值
{
	if (RangeL <= Left && Right <= RangeR) 
	{
		Tree[TreeNode] = (Right - Left + 1) * Val;
		lazy[TreeNode] = Val;
		return;
	}
	ll m = (Left + Right) / 2;
	if (lazy[TreeNode]) 
	{
		Tree[TreeNode<<1] = lazy[TreeNode] * (m - Left + 1);
		Tree[TreeNode<<1| 1] = lazy[TreeNode] * (Right - m);
		lazy[TreeNode<<1] = lazy[TreeNode<<1| 1] = lazy[TreeNode];
		lazy[TreeNode] = 0;
	}
	if (RangeL <= m) bupdate(RangeL, RangeR, Val, Left, m, TreeNode<<1);
	if (RangeR > m) bupdate(RangeL, RangeR, Val, m + 1, Right, TreeNode<<1| 1);
	Tree[TreeNode] = Tree[TreeNode<<1] + Tree[TreeNode<<1| 1];
}
void Mxupdate(const ll& Target, const ll& Val, const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)
{
	if (Left == Right) 
	{
		Tree[TreeNode] = max(Tree[TreeNode], Val);
		return;
	}
	ll Mid = (Left + Right) >> 1;
	if (Target <= Mid) Mxupdate(Target, Val, Left, Mid, TreeNode << 1);
	else Mxupdate(Target, Val, Mid + 1, Right, TreeNode << 1 | 1);
	Tree[TreeNode] = max(Tree[TreeNode << 1], Tree[TreeNode << 1 | 1]);
}
ll Mxquery(const ll& RangeL, const ll& RangeR, const ll& Left=1, const ll& Right=n, const ll& TreeNode=1)
{
	if (RangeL <= Left && Right <= RangeR) return Tree[TreeNode];
	ll Mid = (Left + Right) >> 1,ans=0;
	if (RangeL <= Mid) ans = max(ans, Mxquery(RangeL, RangeR, Left, Mid, TreeNode << 1));
	if (RangeR > Mid) ans = max(ans, Mxquery(RangeL, RangeR, Mid + 1, Right, TreeNode << 1 | 1));
	return ans;
}
